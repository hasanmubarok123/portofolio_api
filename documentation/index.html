Endpoint Public

Method : GET
Request: 
                host/api/public/items
                host/api/public/items/:id
Params: args params 
                p= (integer) : halaman ke
                rp= (integer) : jumlah data per halaman
                id= (integer) : id buku
                judul= (string) : judul buku
                isbn= (integer) : isbn buku
                author= (string) : author buku
                penerbit= (string) : penerbit buku
                kategori_buku.kategori= (string) : kategori buku
                harga= (integer) : harga buku
                stok= (integer) : stok buku
                orderBy= id, isbn, judul, status, harga, stok, penerbit, kategori, createdAt, updatedAt
                sort= asc, desc : default value asc
Params: header 
                token*
Response: 
   
                    error 404:
                    {
                        "message": "Data not found !!!"
                    }

                    success 200:
                    {
                        page: (integer),
                        total_page: (integer),
                        per_page: (integer),
                        data: {
                            id: (integer),
                            judul: (string)
                            isbn: (integer)
                            author: (string)
                            penerbit: (string)
                            kategori_buku.kategori: (string)
                            harga: (integer)
                            stok: (integer,
                            url_picture: (string)
                            status: (string)
                            createdAt: (date string)
                            updatedAt: (date string)
                            users.name: (string)
                        }
                    }
